title Update comment in Post

actor #green:0.5 "Workspace User" as w
boundary "Application \nClient" as f
control "Comment \nController" as gc
participant "Post \nService" as ps
participant "Comment \nService" as gs
participant "Comment \nService" as gs

database #blue:1 "Database" as db

w->f:Update comment in specific post
activate f

f->gc:PUT: /api/comment/{id}
activate gc
gc->gc: check comment's owner
activate gc
gc<--gc: is comment owner
deactivate gc
alt If Current User does not owner of the comment
f<--gc:<color:#red>403 FORBIDDEN</color>
w<--f:Update fail
else 

gc->ps: isExist(id)
activate ps
ps->db: isExist(id)
activate db
ps<--db:Boolean value
deactivate db
gc<--ps:Boolean value
deactivate ps

alt If Post does not exist
f<--gc:<color:#red>404 NOT FOUND: not found</color>
w<--f:Message cannot update comment in post
else 
gc->gs: save(id)
activate gs
gs->db: save(id)
activate db
gs<--db:Comment data
deactivate db
gc<--gs:Comment data
deactivate gs
activate gs
deactivate gs


f<--gc:<color:#green>200 OK: Comment data</color>
deactivate gc

w<--f: Display post with updated comment
deactivate f
end 

end
