boundary "FUniverse Admin \nClient" as f

participant "Combo \nService" as cs
database #blue:1 "Database" as db

activate f

activate ss

deactivate ss

activate gc
gc->gs:get(id)
activate gs
gs->db: get(id)
activate db

deactivate db

deactivate gs

alt If Curriculum empty

else

activate cs

activate db

deactivate gc

deactivate f
end
