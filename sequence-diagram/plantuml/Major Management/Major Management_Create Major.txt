 title Create Major

actor #green:0.5 "Workspace Admin" as w
participant "FUniverse Admin Client" as f
participant "Major Controller" as gc
participant "Major Service" as gs
database #blue:1 "Database" as db

w->f:Click Create Major button
activate f
w<--f:Display Create Major popup
w->f:Type information of Major
f->f:Validate input data
alt If the input data is wrong
w<--f:Validate Message
end
w->f: Click button Save
f->gc:POST: /api/major
activate gc
alt If there is required field do not have value
f<--gc:<color:#red>400 BAD_REQUEST: \nMissing required property</color>
w<--f:Message is required
else
activate gs
gc->gs:save()
gs->db:Create Major
activate db
gs<--db:Major entity
deactivate db
gc<--gs:New Major data
deactivate gs
f<--gc:<color:#green>201 OK: CREATED</color>
deactivate gc
w<--f: Display Major list table with new Major
deactivate f
end
