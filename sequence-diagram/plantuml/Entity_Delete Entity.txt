title Deactive Entity

actor #green:0.5 "Workspace Admin" as w
participant "FUniverse Admin Client" as f
participant "Entity Controller" as gc
participant "Entity Service" as gs
database #blue:1 "Database" as db

w->f:Go to Entity list page
activate f
w->f:Click Delete button
w<--f:Display delete Entity \nconfirmation popup
w->f:Click Deactive button
f->gc:DELETE: /api/{entity}/{id}
activate gc
gc->gs:isExist(id)
activate gs
gs->db:existsById(id)
activate db
gs<--db: Boolean value
deactivate db
gc<--gs:Boolean value
deactivate gs
alt If Entity is not exist
f<--gc:<color:#green>404 NOT FOUND</color>
w<--f: Message Delete fail
else
gc->gs:inactivate(id)
activate gs
gs->db:updateIsActive(id, false)
activate db
gs<--db:
deactivate db
gc<--gs:
deactivate gs
f<--gc:<color:#green>200 OK\nResponse body: Entity data </color>
deactivate gc
w<--f: Message Delete Entity successfully
w<--f: Display Entity list page
deactivate f
end
