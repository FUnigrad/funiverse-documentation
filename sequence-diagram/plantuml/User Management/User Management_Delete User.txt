title Delete User

actor #green:0.5 "Workspace Admin" as w
participant "FUniverse Admin Client" as f
participant "User Controller" as gc
participant "User Service" as gs
database #blue:1 "Database" as db

w->f:Click Delete button
activate f
w<--f:Display delete User \nconfirmation popup
w->f:Click Deactive button
f->gc:DELETE: /api/user/{id}
activate gc
gc->gs:isExist(id)
activate gs
gs->db:existsById(id)
activate db
gs<--db: Boolean value
deactivate db
gc<--gs:Boolean value
deactivate gs
alt If user is not exist
f<--gc:<color:#green>404 NOT FOUND</color>
f->f:Close popup
w<--f: Message Delete fail
else
gc->gs:inactivate(id)
activate gs
gs->db:updateIsActive(id, false)
activate db
gs<--db:
deactivate db
gc<--gs:
deactivate gs
f<--gc:<color:#green>200 OK\nResponse body: User data </color>
deactivate gc
f->f:Close popup
f->f:Update User list page
w<--f: Message Delete User successfully
deactivate f
end
end f
