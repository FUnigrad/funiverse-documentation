title Remove Syllabus from Combo


actor #green:0.5 "Workspace Admin" as w
boundary "FUniverse Admin \nClient" as f
control "Group Controller" as gc
participant "Group Service" as gs
database #blue:1 "Database" as db
w->f:Go to Combo detail page
activate f
w->f:Click delete icon in \nSyllabus list table
w<--f:Remove Syllabus from \nCombo confirmation popup
else
w->f:Click Remove button
f->gc:PUT:/api/combo
activate gc
gc->gs:get(id)
activate gs
gs->db: get(id)
activate db
gs<--db:Combo
deactivate db
gc<--gs: Combo
deactivate gs
gc->gc: Check combo \nisEmpty()
alt If Combo empty

f<--gc:<color:#red>404 NOT FOUND</color>

w<--f: Display screen 404 not found
else
gc->gs:update()
activate gs
gs->db:update()
activate db
gs<--db:
deactivate db
gc<--gs:
deactivate gs
f<--gc:<color:#green>200 OK</color>
deactivate gc
w<--f: Display Combo detail page
deactivate f
end
