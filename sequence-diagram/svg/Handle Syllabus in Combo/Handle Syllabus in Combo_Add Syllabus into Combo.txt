title Add Syllabus into Combo


actor #green:0.5 "Workspace Admin" as w
boundary "FUniverse Admin \nClient" as f
control "Search\nController" as ss
control "Group \nController" as gc
participant "Group \nService" as gs
database #blue:1 "Database" as db
w->f:Go to Combo detail page
activate f
w->f:Click Add Syllabus button
ref over f:Here we have\nexternal process
w<--f:Add Syllabus into Combo popup
w->f:Search Syllabus
f->ss:GET\n/api/search?entity=syllabus\n&{paramToSearch}
activate ss
f<--ss:Syllabus list
deactivate ss
alt If there is no syllabus match condition
w<--f:No Options
else
w->f:Select Syllabus and click Save
f->gc:PUT:/api/combo
activate gc
gc->gs:get(id)
activate gs
gs->db: get(id)
activate db
gs<--db:Combo
deactivate db
gc<--gs: Combo
deactivate gs
gc->gc: Check combo \nisEmpty()
alt If Combo empty

f<--gc:<color:#red>404 NOT FOUND</color>

w<--f: Display screen 404 not found
else
gc->gs:update()
activate gs
gs->db:update()
activate db
gs<--db:
deactivate db
gc<--gs:
deactivate gs
f<--gc:<color:#green>200 OK</color>
deactivate gc
w<--f: Display Combo detail page
deactivate f
end 
deactivate f
end
